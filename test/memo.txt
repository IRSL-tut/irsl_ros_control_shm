rb=RobotBuilder()
alength=0.3
size=5
lk_id = 0
joint_settings = { 'density': 1200.0,
                   'InitialJointAngle': 0.0,
                   'JointRange': [-PI, PI],
                   'JointVelocityRange': [-PI*10, PI*10],
                   'JointEffortRange': [-100, 100],
                   'EquivalentRotorInertia': 0.1 }
cds =coordinates(fv(0, 0, alength*0.5))
jcds=coordinates(fv(0, 0, alength))
b=rb.makeBox(alength*0.4, alength*0.4, alength, color=[0.2, 0.7, 0.2])
b.newcoords(cds)
prev_l=rb.createLinkFromShape(name='Root', root=True, density=1200.0)
for i in range(size):
    cds.translate(fv(0, 0, alength))
    b=rb.makeBox(alength*0.4, alength*0.4, alength, color=[0.2, 0.7, 0.2])
    b.newcoords(cds)
    j=rb.createJointShape(jointType=Link.JointType.RevoluteJoint)
    j.newcoords(jcds)
    jcds.translate(fv(0, 0, alength))
    prev_l=rb.createLinkFromShape(name=f'LINK_{lk_id}', parentLink=prev_l, JointId=lk_id, **joint_settings)
    lk_id += 1
